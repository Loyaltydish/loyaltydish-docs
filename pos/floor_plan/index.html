  $productCategories: [ID]
  $tables: [TableInput]

    Note that extraInfo input field should have the following data:

  "size": {
    "h": 75,
    "w": 75
  },
  "position": {
    "x": 285,
    "y": 120
  }
}

    mutation (
  $floorPlanId: ID!
  $tableId: ID!
  $extraInfo: JSONString
){
  createFloorPlanTable(
    input: {
      floorPlanId: $floorPlanId
      tableId: $tableId
      extraInfo: $extraInfo
    }

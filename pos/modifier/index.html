    Create a modifier

    mutation (
  $name: String!
  $price: Decimal!
  $priceCurrency: String
  $outOfStock: Boolean
  $shortDescription: String!
  $description: String!
  $order: Int!
  $groups: [ID]!
){
  createProductComponent(
    input: {
      name: $name
      price: $price
      priceCurrency: $priceCurrency
      outOfStock: $outOfStock
      shortDescription: $shortDescription
      description: $description
      order: $order
      groups: $groups
    }
  ) {
    errors
    success
    modifier {
      id
    }
  }
}

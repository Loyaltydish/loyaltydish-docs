    Add permission to merchant

     mutation (
  $merchantId: ID!
  $permissionId: ID!
) {
  addPermissionToMerchant(
    input: {
      merchantId: $merchantId
      permissionId: $permissionId
    }
  ) {
    success
    errors
  }
}

  "permissionId": 5
}

    Create a role

     mutation (
  $displayName: String!
  $permissions: [ID]
) {
  createRole(
    input: {
      displayName: $displayName
      permissions: $permissions
    }
  ) {
    success
    errors
  }
